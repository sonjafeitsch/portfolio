'&:hover': !disabled
                ? {
                    backgroundColor: softHoverColor[variant],
                }
                : {},
            };
        }
    }

    export const Button = forwardRef<HTMLButtonElement, ButtonProps>(
    (
        {
        fill,
        className,
        size,
        disabled,
        variant,
        children,
        theme = lightTheme,
        ...otherProps
        },
        ref,
    ) => {
        const styles = getButtonStyles({ fill, size, disabled, variant, theme });

        return (
            <button
                data-testid="button-container"
                ref={ref}
                className={cx('button', styles.button, className)}
                disabled={disabled}
                {...otherProps}>
                {children}
            </button>);
        },
    );

    Button.displayName = 'Button';
